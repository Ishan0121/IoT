🟢 Simple Solution — PIR Motion Sensor + LED

⭐ Start Here — one sensor (PIR), one output (LED). Just 2 pins and a digitalRead.
Serial Monitor
Simulates basic digitalRead() and digitalWrite()
simple.ino — Exp 07 (Beginner)
// Simple PIR Motion Detector
// PIR OUT → Pin 2 | LED → Pin 13

const int PIR_PIN = 2;
const int LED_PIN = 13;

void setup() {
  Serial.begin(9600);
  pinMode(PIR_PIN, INPUT);
  pinMode(LED_PIN, OUTPUT);
}

void loop() {
  int motion = digitalRead(PIR_PIN);

  if(motion == HIGH) {
    digitalWrite(LED_PIN, HIGH);   // Turn LED ON
    Serial.println("Motion detected!");
  } else {
    digitalWrite(LED_PIN, LOW);    // Turn LED OFF
  }
  delay(200);
}

#include <DHT.h>

#define DHTPIN 3
#define DHTTYPE DHT11

DHT dht(DHTPIN, DHTTYPE);

void setup() {
  Serial.begin(9600);
  dht.begin();
}

void loop() {
  // Read humidity and temperature
  float h = dht.readHumidity();
  float t = dht.readTemperature();

  Serial.print("Temp: ");
  Serial.print(t);
  Serial.print("°C  |  Humidity: ");
  Serial.print(h);
  Serial.println("%");

  delay(2000); // DHT11 requires 2s between readings
}

const int LDR_PIN = A0;
const int RELAY_PIN = 7;

void setup() {
  Serial.begin(9600);
  pinMode(RELAY_PIN, OUTPUT);
}

void loop() {
  int lightLevel = analogRead(LDR_PIN);
  Serial.print("Light: ");
  Serial.println(lightLevel);

  // If it gets dark (high resistance, high analog value)
  if(lightLevel > 600) {
    digitalWrite(RELAY_PIN, HIGH); // Turn lamp ON
  } else {
    digitalWrite(RELAY_PIN, LOW);  // Turn lamp OFF
  }
  
  delay(500);
}

💾 Advanced Solution — PIR + DHT11 + LDR + MQ-2 + Relay

🚀 Advanced — 4 sensors working together with non-blocking timers, relay auto-control, and gas alarm.
program.ino — Exp 07
#include <DHT.h>

const int PIR_PIN=2, DHT_PIN=3, LDR_PIN=A0, GAS_PIN=A1;
const int RELAY=7, BUZZ=8, LIGHT_LED=9, GAS_LED=10;
const int DARK_THRESH=600, GAS_THRESH=400;
const float HOT_TEMP=35.0, HIGH_HUM=80.0;
const unsigned long NO_MOTION_MS=10000UL;

#define DHTTYPE DHT11
DHT dht(DHT_PIN, DHTTYPE);

unsigned long lastMotionTime=0, lastDHTTime=0, lastPrint=0;
bool relayState=false;

void checkMotion(){
  bool motion=digitalRead(PIR_PIN);
  if(motion){
    lastMotionTime=millis();
    if(!relayState){relayState=true;digitalWrite(RELAY,HIGH);
      Serial.println(F("[PIR] Motion → Relay ON"));}
  } else if(relayState&&millis()-lastMotionTime>NO_MOTION_MS){
    relayState=false;digitalWrite(RELAY,LOW);
    Serial.println(F("[PIR] No motion 10s → Relay OFF"));
  }
}

void readClimate(){
  float temp=dht.readTemperature(), hum=dht.readHumidity();
  if(isnan(temp)||isnan(hum)){Serial.println(F("[DHT] Error"));return;}
  Serial.print(F("[DHT] "));Serial.print(temp,1);Serial.print(F("°C  "));Serial.print(hum,1);
  Serial.print(F("%"));
  if(temp>HOT_TEMP) Serial.print(F("  ⚠TEMP HIGH"));
  if(hum>HIGH_HUM)  Serial.print(F("  ⚠HUMID HIGH"));
  Serial.println();
}

void checkLight(){
  bool dark=analogRead(LDR_PIN)>DARK_THRESH;
  digitalWrite(LIGHT_LED,dark?HIGH:LOW);
}

void checkGas(){
  bool danger=analogRead(GAS_PIN)>GAS_THRESH;
  digitalWrite(GAS_LED,danger?HIGH:LOW);
  digitalWrite(BUZZ,danger?HIGH:LOW);
  if(danger) Serial.println(F("[GAS] ⚠ DANGER!"));
}

void setup(){
  Serial.begin(9600); dht.begin();
  pinMode(PIR_PIN,INPUT);
  int outs[]={RELAY,BUZZ,LIGHT_LED,GAS_LED};
  for(int i=0;i<4;i++){pinMode(outs[i],OUTPUT);digitalWrite(outs[i],LOW);}
  Serial.println(F("=== Smart Home System Ready ==="));
}

void loop(){
  unsigned long now=millis();
  checkMotion();
  if(now-lastDHTTime>2000){lastDHTTime=now;readClimate();}
  checkLight(); checkGas();
  delay(100);
}

↓

checkMotion()
PIR → Relay ON/OFF after 10s timeout
readClimate()
Every 2s: DHT11 temp+humidity

📝 Notes & Extras

⚠️ MQ-2 Warm-Up Time
MQ-2 gas sensor needs 30 seconds of heating before readings stabilize. The heater coil inside must reach operating temperature. Readings during warm-up are unreliable — always add a warm-up delay loop in setup() before starting monitoring.
💡 LDR Voltage Divider
An LDR (Light-Dependent Resistor) changes resistance with light. Pair with 10kΩ fixed resistor: 5V → 10kΩ → A0 → LDR → GND. In bright light: LDR resistance drops → voltage at A0 drops (low reading). In dark: LDR resistance rises → A0 voltage rises (high reading > threshold).
🔧 isnan() for DHT Validation
dht.readTemperature() returns NaN (Not a Number) on failure (bad wiring, timing). Always check isnan(temp) || isnan(hum) before using the values — printing NaN to Serial causes strange output and attempting math with it propagates errors.
🏠 Relay Wiring Safety
Relay module's INPUT (IN pin) connects to Arduino digital pin. The OUTPUT side (COM/NO/NC) switches high-voltage AC or DC. Never touch the AC side while powered. For lab demos, use low-voltage DC loads (LED strip, fan) instead of mains AC.

🔧 Key Functions

dht.readTemperature() / dht.readHumidity()
Read temperature (°C) and humidity (%) from DHT11. Returns NaN on error. DHT11 needs 2s minimum between readings — track timing with millis().
isnan(value)
Returns true if the float value is NaN (Not a Number). Essential for DHT sensor validation — always check before use.
analogRead(pin)
Read analog pin value 0–1023 (0V–5V). Used for LDR and MQ-2 sensors. Compare against threshold constants to determine state.
checkMotion() — custom
Non-blocking PIR handler: sets relay ON immediately when motion detected, starts 10s timer. Turns relay OFF only after 10s of continuous no-motion — prevents rapid cycling.
