🟢 Simple Solution — Basic WiFi + Cloud Upload

⭐ Start Here — connect to WiFi and send ONE temperature value to ThingSpeak. Minimum code to get data into the cloud.
Serial Monitor
Simulates WiFi connection and basic ThingSpeak GET request loop
simple.ino — Exp 10 (Beginner)
// Simple ESP8266 ThingSpeak Upload
// Upload temperature (DHT11 on D4) to 1 field every 15s

#include <ESP8266WiFi.h>
#include <DHT.h>

const char* SSID    = "YOUR_WIFI_SSID";
const char* PASS    = "YOUR_WIFI_PASSWORD";
const char* API_KEY = "YOUR_WRITE_API_KEY";

DHT dht(D4, DHT11);
WiFiClient client;

void setup() {
  Serial.begin(115200);
  dht.begin();
  WiFi.begin(SSID, PASS);
  Serial.print("Connecting");
  while(WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); }
  Serial.println("\nConnected! IP: " + WiFi.localIP().toString());
}

void loop() {
  float temp = dht.readTemperature();
  if(isnan(temp)) { Serial.println("DHT error!"); delay(15000); return; }

  // Build ThingSpeak URL with field1=temperature
  if(client.connect("api.thingspeak.com", 80)) {
    String url = "/update?api_key=" + String(API_KEY) + "&field1=" + String(temp);
    client.println("GET " + url + " HTTP/1.1");
    client.println("Host: api.thingspeak.com");
    client.println("Connection: close");
    client.println();
    Serial.println("Sent: " + String(temp) + "C");
    client.stop();
  }
  delay(15000); // ThingSpeak free tier minimum
}

💾 Advanced Solution — 4 Fields + HTTPClient + Auto-reconnect

🚀 Advanced — 4 sensor fields, ESP8266HTTPClient library, auto-reconnect on WiFi drop, heat index calculation, LED status.
program.ino — Exp 10 (ESP8266)
#include <ESP8266WiFi.h>
#include <ESP8266HTTPClient.h>
#include <WiFiClient.h>
#include <DHT.h>

// ✏ FILL IN YOUR OWN VALUES:
const char* WIFI_SSID = "YOUR_WIFI_SSID";
const char* WIFI_PASS = "YOUR_WIFI_PASSWORD";
const char* API_KEY   = "YOUR_THINGSPEAK_WRITE_API_KEY";

const char* TS_URL = "http://api.thingspeak.com/update";
#define DHT_PIN D4
#define LDR_PIN A0
#define LED_PIN D0
#define DHTTYPE DHT11
DHT dht(DHT_PIN, DHTTYPE);
WiFiClient wifiClient;

const unsigned long INTERVAL=15000UL;
unsigned long lastUpload=0;
long msgID=0;

void connectWiFi(){
  Serial.print(F("Connecting to WiFi"));
  WiFi.begin(WIFI_SSID,WIFI_PASS);
  int tries=0;
  while(WiFi.status()!=WL_CONNECTED&&tries++<30){
    delay(500);Serial.print(F("."));
  }
  if(WiFi.status()==WL_CONNECTED){
    Serial.print(F("\nIP: ")); Serial.println(WiFi.localIP());
  } else {Serial.println(F("\nFailed! Restarting..."));delay(3000);ESP.restart();}
}

bool uploadToThingSpeak(float t,float h,int ldr,float hi){
  if(WiFi.status()!=WL_CONNECTED){connectWiFi();return false;}
  String url=String(TS_URL)
    +"?api_key="+API_KEY
    +"&field1="+String(t,1)
    +"&field2="+String(h,1)
    +"&field3="+String(ldr)
    +"&field4="+String(hi,1);
  HTTPClient http;
  http.begin(wifiClient,url);
  int code=http.GET();
  bool ok=(code==HTTP_CODE_OK&&http.getString().toInt()>0);
  if(ok){msgID++;Serial.print(F("✓ Upload OK #"));Serial.println(msgID);}
  else Serial.println(F("✗ Upload failed"));
  http.end(); return ok;
}

void setup(){
  Serial.begin(115200);
  pinMode(LED_PIN,OUTPUT); digitalWrite(LED_PIN,HIGH);
  dht.begin(); connectWiFi();
  Serial.println(F("=== ThingSpeak Logger Ready ==="));
}

void loop(){
  unsigned long now=millis();
  if(now-lastUpload>=INTERVAL){
    lastUpload=now;
    float t=dht.readTemperature(),h=dht.readHumidity();
    if(isnan(t)||isnan(h)){Serial.println(F("DHT error"));return;}
    float hi=dht.computeHeatIndex(t,h,false);
    int ldr=analogRead(LDR_PIN);
    Serial.print(F("T="));Serial.print(t,1);Serial.print(F("°C H="));
    Serial.print(h,1);Serial.print(F("% L="));Serial.println(ldr);
    digitalWrite(LED_PIN,LOW); delay(100); digitalWrite(LED_PIN,HIGH);
    uploadToThingSpeak(t,h,ldr,hi);
  }
  yield(); // Keep ESP8266 WiFi stack healthy!
}

📝 Notes & Extras

🔐 Never Commit Credentials!
Your WiFi password and ThingSpeak API key should NEVER be uploaded to a public Git repository. Use a config.h file added to .gitignore, or store credentials in EEPROM at flash time. In production systems, use environment variables or a hardware secure element.
💡 yield() is Critical on ESP8266
yield() gives execution time back to the ESP8266 WiFi stack (LwIP). Without it in tight loops, the WiFi connection degrades and eventually drops. Unlike Arduino UNO, ESP8266 runs WiFi in software — it needs periodic CPU time to maintain the connection.
⏱️ 15-Second Rate Limit
ThingSpeak free tier enforces a 15-second minimum between updates per channel. Uploading faster returns "0" (rate limit). The sketch uses millis()-lastUpload >= 15000 non-blocking timer to respect this limit. Paid tiers allow faster updates.
🌐 HTTP vs MQTT
This experiment uses HTTP REST (GET request). For two-way, real-time IoT, MQTT is better: lightweight publish/subscribe, much lower overhead, supports device-to-device communication. ThingSpeak supports both. MQTT is the industry standard for professional IoT systems.

🔧 Key Functions

WiFi.begin(ssid, password)
Start WiFi connection. Non-blocking — check WiFi.status() == WL_CONNECTED in a loop. Use retry limit to avoid infinite hang.
http.begin(wifiClient, url)
Initialize HTTP client with URL. The WiFiClient handles TCP connection. URL includes API key and field values as query parameters.
http.GET()
Send HTTP GET request. Returns HTTP status code (200=OK). ThingSpeak returns the message ID as a string in the body — "0" means error or rate limited.
dht.computeHeatIndex(temp, hum, false)
Calculate heat index (feels-like temperature) from temperature + humidity. Last parameter: false=Celsius, true=Fahrenheit. Uses Steadman formula.
yield()
ESP8266-specific: temporarily yield CPU to WiFi background tasks. Call at least every 50ms in busy loops. Arduino UNO doesn't need this — only ESP8266/ESP32.
ESP.restart()
Software reset the ESP8266. Used as recovery when WiFi fails after maximum retries — cleaner than getting stuck in an error state.
